(function(angular) {
    "use strict";
 
    var App = App || {};
    App = angular.extend({}, App, {services:{}});
 
    App.services.TmProfileProgramsService = function (
        $resource,
        $q,
        $log,
        constants,
        RequestsQueueService,
        TuProfileService
    ) {
        var service = {
            deferred : {},
            departmentsList: {},
            hasErrors: false
        };
 
        /**
         * Get programs list API endpoint.
         *
         * @returns {$resource}
         */
        service.getProgramsListModel = function () {
            return $resource(constants.api.institutions.url + '/v1/list/tm-programs/core-id/:coreId', {}, {
                get: {
                    cancellable : true
                }
            });
        };

         * Format programs list.
         *
         * @param {Object} programsData
         * @returns {Object}
         */
        service.formatList = function (programsData) {
            // @todo
            return programsData;
        };
 
        service.getAllProgramsByCoreId = function (coreId) {
            if (!coreId) {
                var deferred = $q.defer();
                deferred.resolve(false);
                return deferred.promise;
            }
            if (constants.dev) {
                var startTime = new Date().getTime(),
                    endTime;
            }
 
            return service.getProgramsListModel().get(
                {
                    coreId: coreId
                }
            ).$promise.then(function (data) {
                if (constants.dev) {
                    endTime = new Date().getTime();
                    $log.log('success, got data: ', data, 'in ' + (endTime - startTime) + ' ms');
                }
                return data.toJSON();
            }, function (error) {
                if (constants.dev) {
                    $log.log('request failed' + error);
                }
                return false;
            });
        };
 
        return service;
    };

            App.services.TmProfileProgramsService
        ]);
 
}(window.angular));
