(function(angular) {
    "use strict";
 
    var App = App || {};
    App = angular.extend({}, App, {services:{}});
 
    App.services.InstitutionsService = function ($resource, $q, $log, constants, InstitutionFactory) {
        var service = {
            institutionCoreId: null,
            deferred: {
                getInstitutionData: null
            }
        };
 
        /**
         * Get institutions list API endpoint.
         *
         * @returns {$resource}
         */
        service.getInstitutionListModel = function () {
            return $resource(constants.api.institutions.url + '/v1/list/all/:id');
        };
 
        service.getInstitutionData = function (coreId) {
            if (!coreId) {
                return false;
            }
            coreId = parseInt(coreId, 10);
 
            // process already running
            if (!service.deferred.getInstitutionData) {
                service.deferred.getInstitutionData = $q.defer();
                service.institutionCoreId = coreId;
            } else {
                return service.deferred.getInstitutionData.promise;
            }
            var deferred = service.deferred.getInstitutionData;
 
            var InstitutionListModel = service.getInstitutionListModel();
            InstitutionListModel.get(
                {
                    'filter[coreId]': coreId,
                    limit: 1
                },
                function (data) {
                    if (constants.dev) {
                        $log.log('success, got data: ', data);
                    }
                    service.institutionData = data;
                    deferred.resolve(service.institutionData);
                    service.deferred.getInstitutionData = null;
                }, function (error) {
                    //@todo log error
                    if (constants.dev) {
                        $log.log('request failed' + error);
                    }
                    service.institutionData = [];
                    deferred.resolve(service.institutionData);
                    service.deferred.getInstitutionData = null;
                }
            );
 
            return deferred.promise;
        };
 
        return service;
    };
 
    /**
     * Requests queues.
     *
     * @constructor
     */
    App.services.RequestsQueue = function () {
        var queue = [],
            defaultQueue = 'default';
 
        /**
         * Add request to the queue.
         *
         * @param {Object} request
         * @param {null|String} queueName
         */
        this.add = function(request, queueName) {
            queueName = queueName || defaultQueue;
            if (!queue[queueName]) {
                queue[queueName] = [];
            }
            queue[queueName].push(request);
        };
 
        /**
         * Cancel queue requests.
         *
         * @param {null|String} queueName
         * @returns {boolean}
         */
        this.cancelAll = function (queueName) {
            queueName = queueName || defaultQueue;
            if (!queue[queueName]) {
                return false;
            }
            queue[queueName].forEach(function (request) {
                request.$cancelRequest();
            });
            delete queue[queueName];
        };
    };
 
    angular
        .module('qsHub')
        .service('InstitutionsService', [
            '$resource',
            '$q',
            '$log',
            'constants',
            'InstitutionFactory',
            App.services.InstitutionsService
        ])
        .service('RequestsQueueService', [
            App.services.RequestsQueue
        ]);
}(window.angular));
